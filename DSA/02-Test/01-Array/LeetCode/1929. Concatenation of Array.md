```
class Solution:
    def getConcatenation(self, nums: List[int]) -> List[int]:
        a = []
        first_len = len(nums)
        second_len = 2 * first_len
        for i in range(0,second_len):
            if i < first_len:
                a.append(nums[i])
            else:
                k = i - first_len
                a.append(nums[k])
        return a
```
