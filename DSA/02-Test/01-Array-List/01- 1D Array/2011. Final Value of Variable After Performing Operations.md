```
class Solution:
    def finalValueAfterOperations(self, operations: List[str]) -> int:
        X=0
        total_sum=0
        for op in operations:
            if op == "--X" or op == "X--":
                total_sum = total_sum - 1
            if op == "++X" or op == "X++":
                total_sum = total_sum + 1
        return total_sum
```
